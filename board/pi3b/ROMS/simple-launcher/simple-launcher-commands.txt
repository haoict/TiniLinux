Pico-8 (native)
if [ ! -e "/roms/pico-8/pico8_64" ] || [ ! -e "/roms/pico-8/pico8.dat" ]; then echo "Put pico8_64 and pico8.dat to /roms/pico-8/ to run native pico8" >>/dev/tty0; sleep 5; else chmod +x /roms/pico-8/pico8_64; /usr/local/bin/freqfunctions.sh performance && SDL_RENDER_SCALE_QUALITY=nearest /roms/pico-8/pico8_64 -splore -home /roms/pico-8 && /usr/local/bin/freqfunctions.sh powersave; fi
